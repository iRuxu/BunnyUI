Pagination

< Source Code >
1.引入JS文件 
import Pagination from './$path/Pagination.js'
2.创建实例 
var test = new Pagination({
    container : '.demo',    //加载容器
    total : 28,             //总页数
    limit : 10,             //每组最多显示页数
    start : 2               //起始页
})
3.实例方法 
test.onchange(callback(page))
//当发生页码变更时执行一个回调函数，page为新的页码数，this为当前点击的DOM元素，return返回当前实例
//自行设置项：在数据loading开始之后显示load元素，在数据loaded后隐藏load元素

test.to(page)
//如有需要，可通过该方法绑定外部操作页面跳转

test.getCurrentPage()
//返回当前所在的页数
4.引入CSS文件或自定义外观 
<link rel="stylesheet" href="./$path/Pagination.css" />
